                const nameEl = document.getElementById('dessertName');
                
                showcase.classList.remove('opacity-50', 'scale-95', 'bg-earth-card/30');
                showcase.classList.add('opacity-100', 'scale-100', 'bg-white/60');
                
                iconEl.style.transform = 'scale(0.5)';
                iconEl.style.opacity = '0';
                
                setTimeout(() => {
                    iconEl.textContent = icon;
                    nameEl.textContent = name;
                    
                    iconEl.style.transform = 'scale(1.2)';
                    iconEl.style.opacity = '1';
                    
                    setTimeout(() => {
                        iconEl.style.transform = 'scale(1)';
                    }, 200);
                }, 250);
            };

            // 加入購物車邏輯
            window.addToCart = function(name, price) {
                cart.push({ name, price });
                updateCartUI();
                showToast(`成功加入購物車：${name}`);
            };

            // 更新購物車按鈕與徽章
            function updateCartUI() {
                const cartBtnWrapper = document.getElementById('cartBtnWrapper');
                const cartBadge = document.getElementById('cartBadge');
                
                cartBadge.textContent = cart.length;
                
                if (cart.length > 0) {
                    cartBtnWrapper.classList.remove('hidden');
                } else {
                    cartBtnWrapper.classList.add('hidden');
                }
            }

            // 前往結帳頁面
            window.goToCheckout = function() {
                if (cart.length === 0) {
                    showToast('購物車還是空的喔！');
                    return;
                }
                
                renderCartItems();
                
                const menuSection = document.getElementById('menuSection');
                const checkoutSection = document.getElementById('checkoutSection');
                const cartBtnWrapper = document.getElementById('cartBtnWrapper');
                
                menuSection.classList.add('opacity-0');
                cartBtnWrapper.classList.add('hidden'); // 在結帳頁隱藏浮動按鈕
                
                setTimeout(() => {
                    menuSection.classList.add('hidden');
                    checkoutSection.classList.remove('hidden');
                    void checkoutSection.offsetWidth;
                    checkoutSection.classList.remove('opacity-0');
                    window.scrollTo(0, 0);
                }, 500);
            };

            // 渲染結帳頁的餐點清單
            function renderCartItems() {
                const listContainer = document.getElementById('cartItemsList');
                const totalContainer = document.getElementById('cartTotal');
                let html = '';
                let total = 0;

                cart.forEach((item, index) => {
                    total += item.price;
                    html += `
                        <div class="flex justify-between items-center bg-earth-input p-3 rounded-lg border border-earth-border/50">
                            <span class="tracking-wide">${item.name}</span>
                            <span class="font-medium opacity-80">$${item.price}</span>
                        </div>
                    `;
                });

                listContainer.innerHTML = html;
                totalContainer.textContent = `$${total}`;
            }

            // 繼續點餐 (返回選單)
            window.backToMenu = function() {
                const menuSection = document.getElementById('menuSection');
                const checkoutSection = document.getElementById('checkoutSection');
                
                checkoutSection.classList.add('opacity-0');
                
                setTimeout(() => {
                    checkoutSection.classList.add('hidden');
                    menuSection.classList.remove('hidden');
                    void menuSection.offsetWidth;
                    menuSection.classList.remove('opacity-0');
                    updateCartUI(); // 恢復購物車按鈕顯示
                    window.scrollTo(0, 0);
                }, 500);
            };

            // 監聽集點 Checkbox 控制手機輸入框顯示
            pointsCheckbox.addEventListener('change', (e) => {
                if (e.target.checked) {
                    phoneInputWrapper.classList.remove('hidden');
                } else {
                    phoneInputWrapper.classList.add('hidden');
                }
            });

            // 完成訂單
            window.completeOrder = function() {
                // 檢查是否勾選集點但未輸入電話
                if (pointsCheckbox.checked) {
                    const phone = document.getElementById('phoneNumber').value;
                    if (!phone.trim()) {
                        showToast('請輸入手機號碼以便為您集點喔！');
                        return;
                    }
                }

                const checkoutSection = document.getElementById('checkoutSection');
                const successSection = document.getElementById('successSection');

                checkoutSection.classList.add('opacity-0');

                setTimeout(() => {
                    checkoutSection.classList.add('hidden');
                    successSection.classList.remove('hidden');
                    void successSection.offsetWidth;
                    successSection.classList.remove('opacity-0');
                    window.scrollTo(0, 0);
                    
                    // 清空購物車
                    cart = [];
                    updateCartUI();
                }, 500);
            };
        });
    </script>
